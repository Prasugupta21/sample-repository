#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    
    int n,q,s=0,num;
    cin>>n>>q;
    int *a[n];
    while(n--)
    {
        cin>>num;
        a[s]=new int [num];
        for(int i=0;i<num;i++)
        {
            cin>>a[s][i];
        }
        s++;
    }
    while(q--)
    {
        int x,y;
        cin>>x>>y;
        cout<<a[x][y]<<endl;
    }
    
    return 0;
}
