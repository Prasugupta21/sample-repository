
 bool xend(vector<int>&a,vector<int>&b){
             return a[1]<b[1];
         }
class Solution {
public:
    int findMinArrowShots(vector<vector<int>>& p) {
        

        sort(p.begin(),p.end(),xend);
        int arrow=1;
        int end=p[0][1];
        for(int i=1;i<p.size();i++){
            if(p[i][0]>end){
                arrow++;
                end=p[i][1];
            }
        }
        return arrow;
    }
};
