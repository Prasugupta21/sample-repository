class Solution {
public:
    char findTheDifference(string s, string t) {
        sort(s.begin(),s.end());
         sort(t.begin(),t.end());
        int i=0,j=0;
        while(i<s.size()){
            if(s[i]!=t[i]){
                return  t[i];
            }
            i++;
        }
        if(i==s.size()){

        return t[t.size()-1];
            }
            return char (0);
    }

};
