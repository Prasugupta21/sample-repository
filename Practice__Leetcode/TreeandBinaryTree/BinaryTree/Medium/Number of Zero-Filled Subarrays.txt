class Solution {
public:
    long long zeroFilledSubarray(vector<int>& nums) {
      long long cnt=0;
      long long ZeroSubArrayEndAtCurrIdx=0;
      for(auto  x : nums){
        if(!x){
          cnt+=++ZeroSubArrayEndAtCurrIdx;
        }
        else{
          ZeroSubArrayEndAtCurrIdx=0;
        }
      }
      return cnt;
    }
};
